 25    def _ensure_database_exists(self):
 26        """
 27        Ensures that the target PostgreSQL database exists.
 28        If it doesn't, attempts to connect to the default 'postgres' database
 29        and issues a CREATE DATABASE command to create it.
 30        """
 31        target_db = self.db_config.get('database')
 32        try:
 33            # Try connecting to the target database to see if it exists
 34            conn = psycopg2.connect(
 35                user=self.db_config.get('user'),
 36                password=self.db_config.get('password'),
 37                host=self.db_config.get('host'),
 38                port=self.db_config.get('port'),
 39                database=target_db
 40            )
 41            conn.close()
 42        except psycopg2.OperationalError as e:
 43            if f'database "{target_db}" does not exist' in str(e):
 44                print(f"Database '{target_db}' not found. Attempting to create it...")
 45                try:
 46                    # Connect to 'postgres' database to create the new one
 47                    conn = psycopg2.connect(
 48                        user=self.db_config.get('user'),
 49                        password=self.db_config.get('password'),
 50                        host=self.db_config.get('host'),
 51                        port=self.db_config.get('port'),
 52                        database='postgres'
 53                    )
 54                    conn.autocommit = True
 55                    with conn.cursor() as cursor:
 56                        cursor.execute(f'CREATE DATABASE {target_db}')
 57                    conn.close()
 58                    print(f"Database '{target_db}' created successfully.")
 59                except Exception as create_error:
 60                    print(f"Failed to create database '{target_db}': {create_error}")
 61            else:
 62                print(f"Operational error while checking database: {e}")
 63        except Exception as e:
 64            print(f"Error while checking database existence: {e}")

 66    def _initialize_pool(self):

 68        Initializes a psycopg2 SimpleConnectionPool.
 69        Establishes a minimum of 1 and maximum of 20 concurrent connections
 70        using the provided database credentials.
 71        """
 72        try:

 74                1, 20,

 78                port=self.db_config.get('port'),

 82                print("Connection pool created successfully")

305    def execute_query(self, query, params=None):
306        """
307        Executes a SQL query that modifies data (INSERT, UPDATE, DELETE).
308        Automatically acquires a connection from the pool, commits the transaction,
309        and returns the cursor upon success. Rolls back the transaction on error.
310        """
311        if not self.connection_pool:
312            return None
313        conn = self.connection_pool.getconn()
314        try:
315            with conn.cursor() as cursor:
316                cursor.execute(query, params)
317                conn.commit()
318                return cursor
319        except (Exception, psycopg2.DatabaseError) as error:
320            print(f"Error executing query: {error}")
321            conn.rollback()
322        finally:
323            if self.connection_pool:
324                self.connection_pool.putconn(conn)

326    def fetch_one(self, query, params=None):
327        """
328        Executes a SELECT query and returns a single row (tuple).
329        Safely acquires and releases a connection from the pool.
330        """

332            return None
333        conn = self.connection_pool.getconn()
334        try:

336                cursor.execute(query, params)

339            print(f"Error fetching data: {error}")
340        finally:
341            if self.connection_pool:

398        """
399        Inserts or updates the extended profile details of a user (e.g., pronouns, 
400        favorite colors, hobbies, known technologies). Accepts a dynamic set of fields via kwargs.
401        """
402        if not kwargs:
403            return
404        
405        columns = ['user_id'] + list(kwargs.keys())
406        values = [user_id] + list(kwargs.values())
407        
408        placeholders = ', '.join(['%s'] * len(columns))
409        col_names = ', '.join(columns)
410        
411        updates = ', '.join([f"{k} = EXCLUDED.{k}" for k in kwargs.keys()])
412        
413        query = f"""
414        INSERT INTO user_profiles ({col_names})
415        VALUES ({placeholders})
416        ON CONFLICT (user_id) DO UPDATE SET
417            {updates}
418        """
419        
420        try:
421            self.execute_query(query, tuple(values))
422            print(f"[DB] User profile {user_id} updated successfully.")
423        except Exception as e:

 31        """

 46                try:

306    def execute_query(self, query, params=None):
307        """
308        Executes a SQL query that modifies data (INSERT, UPDATE, DELETE).
309        Automatically acquires a connection from the pool, commits the transaction,
310        and returns the cursor upon success. Rolls back the transaction on error.
311        """
312        if not self.connection_pool:
313            return None
314        conn = self.connection_pool.getconn()
315        try:
316            with conn.cursor() as cursor:
317                cursor.execute(query, params)
318                conn.commit()
319                return cursor
320        except (Exception, psycopg2.DatabaseError) as error:
321            print(f"Error executing query: {error}")
322            conn.rollback()
323        finally:
324            if self.connection_pool:
325                self.connection_pool.putconn(conn)

307        """
308        Executes a SQL query that modifies data (INSERT, UPDATE, DELETE).
309        Automatically acquires a connection from the pool, commits the transaction,
310        and returns the cursor upon success. Rolls back the transaction on error.
311        """
312        if not self.connection_pool:
313            return None
314        conn = self.connection_pool.getconn()
315        try:
316            with conn.cursor() as cursor:
317                cursor.execute(query, params)
318                conn.commit()
319                return cursor
320        except (Exception, psycopg2.DatabaseError) as error:
321            print(f"Error executing query: {error}")
322            conn.rollback()
323        finally:
324            if self.connection_pool:
325                self.connection_pool.putconn(conn)
