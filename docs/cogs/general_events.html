  6class GeneralEventsCog(commands.Cog):
  7    """
  8    Discord Cog responsible for catching and logging a wide variety of general server events
  9    (reactions, channel/role changes, bans, timeouts, message deletions) into the database.
 10    """
 11    def __init__(self, client, config, database):
 12        """
 13        Initializes the GeneralEventsCog with the bot, configuration, and database instances.
 14        """
 15        self.client = client
 16        self.config = config
 17        self.database = database
 18        self.tz = ZoneInfo("Europe/Warsaw")
 19
 20    def _log_event(self, user_id, user_name, what, about, channel=None, guild=None, is_bot=False):
 21        """
 22        Core helper method to format and insert an event into the database.
 23        It respects configured blacklists for both channels and user roles before logging.
 24        """
 25        guild_id = guild.id if guild else (channel.guild.id if channel else None)
 26        channel_id = channel.id if channel else None
 27
 28        if guild_id:
 29            # Check if channel is blacklisted
 30            if channel_id and self.database.is_blacklisted(guild_id, channel_id):
 31                return
 32
 33            # Check if user roles are blacklisted
 34            if user_id:
 35                guild_obj = guild or (channel.guild if channel else self.client.get_guild(guild_id))
 36                if guild_obj:
 37                    member = guild_obj.get_member(user_id)
 38                    if member:
 39                        for role in member.roles:
 40                            if self.database.is_blacklisted(guild_id, role.id):
 41                                return
 42
 43        channel_name = channel.name if channel else None
 44        category_id = channel.category.id if channel and hasattr(channel, 'category') and channel.category else None
 45        category_name = channel.category.name if channel and hasattr(channel, 'category') and channel.category else None
 46        guild_name = guild.name if guild else (channel.guild.name if channel else None)
 47
 48        # Current time in Polish timezone
 49        now = datetime.now(self.tz)
 50
 51        self.database.put_event(
 52            user_id, user_name, is_bot, what, about, now, channel_id, channel_name,
 53            guild_id, guild_name, category_id, category_name
 54        )
